    src/components/ApplyPage/ApplyNowPage.js

    import React from 'react';
import PropTypes from 'prop-types';
import './styles.scss';
import { connect } from 'react-redux';
import i18n from 'i18n-js';
import ApplyNow1 from './ApplyNow1';
import ApplyNow2 from './ApplyNow2';
import ApplyNow3 from './ApplyNow3';
import ApplyNow4 from './ApplyNow4';
import ApplyNow5 from './ApplyNow5';
import Congrads from './Congrads';
import * as funcsFromStore from '../../store/store';

/**
 * @desc container for UI components of candidate registration screen
 * @param {string} language
 * @returns {JSX.Element}
 */
const ApplyNowPage = ({ language }) => {
	i18n.locale = language;

	return (
		<article className='applyNow'>
			<ApplyNow1 />
			<ApplyNow2 />
			<ApplyNow3 />
			<ApplyNow4 />
			<ApplyNow5 />
			<Congrads />
		</article>
	);
};

const storeFuncs = {
	applyLanguage: funcsFromStore.applyLanguage,
};

const storeData = (state) => ({
	language: state.language,
});

ApplyNowPage.propTypes = {
	language: PropTypes.string.isRequired,
};

export default connect(storeData, storeFuncs)(ApplyNowPage);
