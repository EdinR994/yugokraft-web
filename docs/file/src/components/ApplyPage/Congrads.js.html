    src/components/ApplyPage/Congrads.js

    import './styles.scss';
import React from 'react';
import i18n from 'i18n-js';
import { Link } from 'react-router-dom';
import { connect } from 'react-redux';
import congrats from './images/congrats.svg';
import routes from '../../constants/routes';
import * as funcsFromStore from '../../store/store';

/**
 * @desc UI for screen after registration success
 * @param {string} languageSelected
 * @param {object} displayApplyNow
 * @param {function} changeDisplayApplyNow
 * @param {function} applyUser
 * @returns {JSX.Element}
 */
const Congrads = ({ languageSelected, displayApplyNow, changeDisplayApplyNow, applyUser }) => {
	i18n.locale = languageSelected;

	if (!displayApplyNow[5]) return null;
	window.scrollTo({ top: 0, behavior: 'smooth' });
	localStorage.setItem('yugokraftId', '');

	return (
		<section className='applyNow__page'>
			<div className='applyNow__form '>
				<h2 className='applyNow__header'>{i18n.t('You-have-applied-for-job')}</h2>
				<div className='applyNow__congratsImageWrapper'>
					<img src={congrats} className='applyNow__congratsImage' alt={i18n.t('Congratulations')} />
					<p className='applyNow__congratsWord'>{i18n.t('Congratulations')}</p>
					<p className='applyNow__congratsTextUnderWord'>{i18n.t('We-will-consider')}</p>
				</div>
			</div>
			<Link
				to={routes.HOME}
				className='applyNow__home'
				onClick={() => {
					const copyArr = [...displayApplyNow];
					const indexOfDisplayApplyNow = copyArr.indexOf(true);
					copyArr[indexOfDisplayApplyNow] = false;
					copyArr[0] = true;
					changeDisplayApplyNow(copyArr);
					applyUser({});
				}}
			>
				{i18n.t('Home')}
			</Link>
		</section>
	);
};

const storeFuncs = {
	changeDisplayApplyNow: funcsFromStore.applyDisplayApplyNow,
	applyUser: funcsFromStore.applyUser,
};

const storeData = (state) => ({
	displayApplyNow: state.displayApplyNow,
	languageSelected: state.language,
});

export default connect(storeData, storeFuncs)(Congrads);
