    src/components/ApplyPage/ApplyNow3.js

    import './styles.scss';
import React, { useState, useEffect } from 'react';
import { v4 as uuidv4 } from 'uuid';
import i18n from 'i18n-js';
import { connect } from 'react-redux';
import PropTypes from 'prop-types';
import arrowRight from './images/arrowRight.svg';
import plus from './images/plus.svg';
import arrowLeft from './images/arrowLeft.svg';
import * as funcsFromStore from '../../store/store';
import Select from '../Select/Select';
import translator from './translator';

/**
 * @desc UI for candidate registration form - step 3
 * @param {object} displayApplyNow
 * @param {function} changeDisplayApplyNow
 * @param {function} applyUser
 * @param {object} userFromRedux
 * @param {string} languageSelected
 * @returns {JSX.Element|null}
 */
const ApplyNow3 = ({ languageSelected, displayApplyNow, changeDisplayApplyNow, applyUser, userFromRedux }) => {
	const [languageNameInputs, setLanguageNameInputs] = useState([0, 1]);
	const [languageName, setLanguageName] = useState([]);
	const [languageLevel, setLanguageLevel] = useState([]);
	const [additionLanguagesAmount, setAdditionLanguagesAmount] = useState(2);
	const [expierenceAboard, setExpierenceAboard] = useState('');
	const [skillsInputs, setSkillsInputs] = useState([0]);
	const [skills, setSkills] = useState([]);
	const [skillsDescriptionsLevel, setSkillsDescriptionsLevel] = useState([]);
	const [additionSkillsAmount, setAdditionSkillsAmount] = useState(1);
	const [incorrectLanguageLevel, setIncorrectLanguageLevel] = useState([false]);
	const [incorrectExpierence, setIncorrectExpierence] = useState(false);
	const [incorrectSkills, setIncorrectSkills] = useState(false);
	const [levelWithoutLanguage, setLevelWithoutLanguage] = useState([false]);

	const saveToRedux = (userKey, key, index, value) => {
		const languages = [];

		for (let i = 0; i < additionLanguagesAmount; i += 1) {
			languages.push({ languageName: languageName[i], level: languageLevel[i] });
		}

		if (userKey === 'languages') {
			languages[index] = { ...languages[index], [key]: value };
		}

		const copySkills = [];

		for (let i = 0; i < additionSkillsAmount; i += 1) {
			copySkills.push({ skillName: skills[i], description: skillsDescriptionsLevel[i] });
		}

		if (userKey === 'skills') {
			copySkills[index] = { ...copySkills[index], [key]: value };
		}

		const copyExpierenceAboard = key === 'expierenceAboard' ? value : expierenceAboard;

		applyUser({
			...userFromRedux,
			languages,
			expierenceAboard: copyExpierenceAboard,
			skills: copySkills,
		});
	};

	useEffect(() => {
		if (!userFromRedux.skills) {
			applyUser({
				...userFromRedux,
				skills: [{ skillName: '', description: '' }],
			});
		}

		if (!userFromRedux.languages) {
			applyUser({
				...userFromRedux,
				languages: [
					{ languageName: '', level: '' },
					{ languageName: '', level: '' },
				],
			});
		}

		if (typeof userFromRedux.expierenceAboard !== 'string') {
			applyUser({
				...userFromRedux,
				expierenceAboard: '',
			});
		}

		if (userFromRedux.languages && languageName.length !== userFromRedux.languages.length) {
			const copyLanguageName = [];
			const copyLevel = [];

			for (let i = 0; i < userFromRedux.languages.length; i += 1) {
				copyLanguageName.push(userFromRedux.languages[i].languageName);
				copyLevel.push(userFromRedux.languages[i].level);
			}

			setLanguageName(copyLanguageName);
			setLanguageLevel(copyLevel);
			if (userFromRedux.languages.length > 2) {
				const copyLanguageNameInputs = [];

				for (let i = 0; i < userFromRedux.languages.length; i += 1) {
					copyLanguageNameInputs.push(i);
				}

				setAdditionLanguagesAmount(userFromRedux.languages.length);
				setLanguageNameInputs(copyLanguageNameInputs);
			}
		}

		if (userFromRedux.skills && skills.length !== userFromRedux.skills.length) {
			const copySkillName = [];
			const copyDescription = [];
			const copySkillsInputs = [];

			for (let i = 0; i < userFromRedux.skills.length; i += 1) {
				copySkillName.push(userFromRedux.skills[i].skillName);
				copyDescription.push(userFromRedux.skills[i].description);
				copySkillsInputs.push(i);
			}

			setSkills(copySkillName);
			setSkillsDescriptionsLevel(copyDescription);
			setAdditionSkillsAmount(userFromRedux.skills.length);
			setSkillsInputs(copySkillsInputs);
		}

		if (userFromRedux.expierenceAboard && expierenceAboard !== userFromRedux.expierenceAboard) {
			setExpierenceAboard(userFromRedux.expierenceAboard);
		}
	}, [userFromRedux, languageName.length, skills.length, expierenceAboard, applyUser]);

	i18n.locale = languageSelected;

	if (!displayApplyNow[2]) return null;

	const next = () => {
		let correct = true;

		if (!languageLevel.length === 0) {
			const copyIncorrectLanguageLevel = [...incorrectLanguageLevel];
			copyIncorrectLanguageLevel[0] = true;
			setIncorrectLanguageLevel(copyIncorrectLanguageLevel);
			correct = false;
		}

		const copyIncorrectLanguageLevel = [...incorrectLanguageLevel];

		for (let i = 0; i < languageNameInputs.length; i += 1) {
			if (languageName[i] && !languageLevel[i]) {
				copyIncorrectLanguageLevel[i] = true;
				correct = false;
			}
		}

		setIncorrectLanguageLevel(copyIncorrectLanguageLevel);

		const copyLevelWithoutLanguage = [...levelWithoutLanguage];

		for (let i = 0; i < languageNameInputs.length; i += 1) {
			if (!languageName[i] && languageLevel[i]) {
				copyLevelWithoutLanguage[i] = true;
				correct = false;
			}
		}

		setLevelWithoutLanguage(copyLevelWithoutLanguage);

		if (correct) {
			const copyDisplayApplyNow = [...displayApplyNow];
			const indexOfDisplayApplyNow = copyDisplayApplyNow.indexOf(true);
			copyDisplayApplyNow[indexOfDisplayApplyNow] = false;
			copyDisplayApplyNow[indexOfDisplayApplyNow + 1] = true;

			changeDisplayApplyNow(copyDisplayApplyNow);
		}
	};

	let languagesVars = [
		{ id: uuidv4(), name: 'Deutsch', value: 'Deutsch' },
		{ id: uuidv4(), name: 'Englisch', value: 'Englisch' },
		{ id: uuidv4(), name: 'Französisch', value: 'Französisch' },
		{ id: uuidv4(), name: 'Russisch', value: 'Russisch' },
		{ id: uuidv4(), name: 'Italienisch', value: 'Italienisch' },
		{ id: uuidv4(), name: 'Spanisch', value: 'Spanisch' },
		{ id: uuidv4(), name: 'Türksich', value: 'Türksich' },
	];

	languagesVars = languagesVars.filter((lang) => !languageName.includes(lang.value));

	if (!languagesVars.includes({ name: 'language', value: '' })) {
		languagesVars.unshift({ id: uuidv4(), name: 'language', value: '' });
	}

	const languagesLevelVars = [
		{ id: uuidv4(), name: 'level', value: '' },
		{ id: uuidv4(), name: 'A Anfänger', value: 'A Anfänger' },
		{ id: uuidv4(), name: 'B Mittleres', value: 'B Mittleres' },
		{ id: uuidv4(), name: 'C Fortgeschritten', value: 'C Fortgeschritten' },
	];

	let skillsVars = [
		{ id: uuidv4(), name: 'Kommunikation', value: 'Kommunikation' },
		{ id: uuidv4(), name: 'Teamfähigkeit', value: 'Teamfähigkeit' },
		{ id: uuidv4(), name: 'Problemlösung', value: 'Problemlösung' },
		{ id: uuidv4(), name: 'Kreativität', value: 'Kreativität' },
		{ id: uuidv4(), name: 'Arbeitsethik', value: 'Arbeitsethik' },
		{ id: uuidv4(), name: 'Zwischenmenschliche Beziehungen', value: 'Zwischenmenschliche Beziehungen' },
		{ id: uuidv4(), name: 'Zeitmanagement', value: 'Zeitmanagement' },
		{ id: uuidv4(), name: 'Führungsqualität', value: 'Führungsqualität' },
		{ id: uuidv4(), name: 'Präzision und Detailtreue', value: 'Präzision und Detailtreue' },
	];

	skillsVars = skillsVars.filter((skillVar) => !skills.includes(skillVar.value));

	if (!skillsVars.includes({ name: 'choose-your-field-of-activity', value: '' })) {
		skillsVars.unshift({ id: uuidv4(), name: 'choose-your-field-of-activity', value: '' });
	}

	const addLanguage = () => {
		if (additionLanguagesAmount < 7) {
			let copyLanguagesFromRedax = [{ languageName: '', level: '' }];

			if (userFromRedux.languages) {
				copyLanguagesFromRedax = [...userFromRedux.languages];
			}

			copyLanguagesFromRedax.push({ languageName: '', level: '' });

			applyUser({
				...userFromRedux,
				languages: copyLanguagesFromRedax,
			});
		}
	};

	const addSkills = () => {
		if (additionSkillsAmount < 4) {
			let copySkillsFromRedax = [{ skillName: '', description: '' }];

			if (userFromRedux.languages) {
				copySkillsFromRedax = [...userFromRedux.skills];
			}

			copySkillsFromRedax.push({ skillName: '', description: '' });

			applyUser({
				...userFromRedux,
				skills: copySkillsFromRedax,
			});
		}
	};

	const deleteOneElementFromArray = (array, index, setArray) => {
		const copyArray = [...array];
		copyArray.splice(index, 1);

		if (setArray) setArray(copyArray);

		return copyArray;
	};

	const deleteLanguage = (index) => {
		setAdditionLanguagesAmount(additionLanguagesAmount - 1);

		const copyLanguageNameInputs = [...languageNameInputs];
		copyLanguageNameInputs.pop();
		setLanguageNameInputs(copyLanguageNameInputs);
		deleteOneElementFromArray(languageName, index, setLanguageName);
		deleteOneElementFromArray(languageLevel, index, setLanguageLevel);
		deleteOneElementFromArray(incorrectLanguageLevel, index, setIncorrectLanguageLevel);
		deleteOneElementFromArray(levelWithoutLanguage, index, setLevelWithoutLanguage);

		applyUser({
			...userFromRedux,
			languages: deleteOneElementFromArray(userFromRedux.languages, index),
		});
	};

	const deleteSkill = (index) => {
		setAdditionSkillsAmount(additionSkillsAmount - 1);
		const copySkillsInputs = [...skillsInputs];
		copySkillsInputs.pop();
		setSkillsInputs(copySkillsInputs);

		deleteOneElementFromArray(skills, index, setSkills);
		deleteOneElementFromArray(skillsDescriptionsLevel, index, setSkillsDescriptionsLevel);

		applyUser({
			...userFromRedux,
			skills: deleteOneElementFromArray(userFromRedux.skills, index),
		});
	};

	const changeLanguageName = (value, index) => {
		const copyLanguageName = [...languageName];
		copyLanguageName[index] = value;
		setLanguageName(copyLanguageName);
		saveToRedux('languages', 'languageName', index, value);
	};

	const changeLanguageLevel = (value, index) => {
		const copyLanguageLevel = [...languageLevel];
		copyLanguageLevel[index] = value;
		setLanguageLevel(copyLanguageLevel);
		saveToRedux('languages', 'level', index, value);
	};

	const changeSkills = (value, index) => {
		const copySkills = [...skills];
		copySkills[index] = value;
		setSkills(copySkills);
		saveToRedux('skills', 'skillName', index, value);
	};

	const disableLanguagesErrors = (index) => {
		const copyLevelWithoutLanguage = [...levelWithoutLanguage];
		copyLevelWithoutLanguage[index] = false;
		setLevelWithoutLanguage(copyLevelWithoutLanguage);
	};

	const disableLanguageLevelErrors = (index) => {
		const copyIncorrectLanguageLevel = [...incorrectLanguageLevel];
		copyIncorrectLanguageLevel[index] = false;
		setIncorrectLanguageLevel(copyIncorrectLanguageLevel);
	};

	return (
		<section className='applyNow__page'>
			<form className='applyNow__form applyNow__form2'>
				<h2 className='applyNow__header'>{i18n.t('Languages-and-experience-abroad')}</h2>

				{languageNameInputs.map((index, arrayIndex) => {
					if (arrayIndex >= 7) return null;

					return (
						<div key={index} className='applyNow__labelsWrapper'>
							<label htmlFor='_labelHeader'>
								<h3 className={arrayIndex === 0 ? 'applyNow__labelHeader' : 'applyNow__labelHeaderNotReq'}>{i18n.t('language')}</h3>
								<Select
									values={languagesVars}
									cb={(value) => {
										changeLanguageName(value, index);
									}}
									placeHolder={
										translator(languageName[index], (value) => {
											const copyLanguageName = [...languageName];
											copyLanguageName[index] = value;
											setLanguageName(copyLanguageName);
										}) || i18n.t('language')
									}
									selected={!!languageName[index]}
									classWord={levelWithoutLanguage[index] ? 'applyNow__incorect ' : ''}
									click={() => {
										disableLanguagesErrors(index);
									}}
								/>
							</label>

							<label htmlFor='labelHeader' className='applyNow__label'>
								<h3 className='applyNow__labelHeader'>{i18n.t('level')}</h3>
								<Select
									values={languagesLevelVars}
									cb={(value) => {
										changeLanguageLevel(value, index);
									}}
									placeHolder={
										translator(languageLevel[index], (value) => {
											const copyLanguageLevel = [...languageLevel];
											copyLanguageLevel[index] = value;
											setLanguageLevel(copyLanguageLevel);
										}) || i18n.t('level')
									}
									selected={!!languageLevel[index]}
									classWord={
										(incorrectLanguageLevel[index] && languageName[index] && !languageLevel[index] ? 'applyNow__incorect ' : '') +
										(index > 1 ? ' select__short' : '')
									}
									click={() => {
										disableLanguageLevelErrors(index);
									}}
								/>

								{index > 1 ? (
									<div
										role='button'
										onKeyPress={() => {}}
										tabIndex={0}
										className='applyNow__delete'
										onClick={() => {
											deleteLanguage(index);
										}}
									>
										&times;
									</div>
								) : null}
							</label>
						</div>
					);
				})}

				<label htmlFor='_addButton' className='applyNow__addWrapper'>
					<button id='_addButton' type='button' className='applyNow__addButton' onClick={addLanguage}>
						<img src={plus} alt={i18n.t('add-one-more-language')} />
					</button>
					<h3 className='applyNow__addText'>{i18n.t('add-one-more-language')}</h3>
				</label>

				<label htmlFor='_textareaLabel' className='applyNow__textareaLabel'>
					<h3 className='applyNow__labelHeader'>{i18n.t('Experience-abroad')}</h3>
					<textarea
						className={
							incorrectExpierence
								? 'applyNow__textarea applyNow__textareaExpierence applyNow__incorect'
								: 'applyNow__textarea applyNow__textareaExpierence'
						}
						placeholder={expierenceAboard || i18n.t('describe-your-experience-abroad')}
						onFocus={() => setIncorrectExpierence(false)}
						value={expierenceAboard || ''}
						maxLength='1250'
						onChange={(e) => {
							setExpierenceAboard(e.target.value);
							saveToRedux(null, 'expierenceAboard', null, e.target.value);
						}}
					/>
				</label>

				{skillsInputs.map((index) => {
					return (
						<div key={`${index}s`} className='applyNow__labelsWrapper applyNow__skillsWrapper'>
							<label htmlFor='Now__labelHeader'>
								<h3 className='applyNow__labelHeader'>{i18n.t('skills-up-to')}</h3>
								<Select
									values={skillsVars}
									cb={(value) => {
										changeSkills(value, index);
									}}
									placeHolder={
										translator(skills[index], (value) => {
											const copySkills = [...skills];
											copySkills[index] = value;
											setSkills(copySkills);
										}) || i18n.t('choose-your-field-of-activity')
									}
									selected={!!skills[index]}
									classWord={(incorrectSkills ? 'applyNow__incorect ' : '') + (index ? ' select__short' : '')}
									click={() => {
										setIncorrectSkills(false);
									}}
								/>
							</label>

							<label htmlFor='w__textareaLabel' className='applyNow__textareaLabel applyNow__textareaSkillsExpierenceLabel'>
								<h3 className='applyNow__labelHeaderNotReq'>{i18n.t('Description')}</h3>
								<div className={index ? 'applyNow__labelTextArea applyNow__labelTextAreaShort' : 'applyNow__labelTextArea'}>
									<textarea
										className='applyNow__textarea applyNow__textareaSkillsExpierence'
										placeholder={i18n.t('Describe-your-skills-and-experience')}
										value={skillsDescriptionsLevel[index] || ''}
										maxLength='1250'
										onChange={(e) => {
											const copySkillsDescriptionsLevel = [...skillsDescriptionsLevel];
											copySkillsDescriptionsLevel[index] = e.target.value;
											setSkillsDescriptionsLevel(copySkillsDescriptionsLevel);
											saveToRedux('skills', 'description', index, e.target.value);
										}}
									/>
									{index ? (
										<div
											role='button'
											onKeyPress={() => {}}
											tabIndex={0}
											className='applyNow__delete'
											onClick={() => {
												deleteSkill(index);
											}}
										>
											&times;
										</div>
									) : null}
								</div>
							</label>
						</div>
					);
				})}

				<label htmlFor='ow__addWrapper' className='applyNow__addWrapper'>
					<button id='ow__addWrapper' type='button' className='applyNow__addButton' onClick={addSkills}>
						<img src={plus} alt={i18n.t('Add-one-more-field')} />
					</button>
					<h3 className='applyNow__addText'>{i18n.t('Add-one-more-field')}</h3>
				</label>
			</form>

			<div className='applyNow__labelsWrapper applyNow__ButtonsWrapper2'>
				<button
					type='button'
					className='applyNow__backButton'
					onClick={() => {
						const copyDisplayApplyNow = [...displayApplyNow];
						const indexOfDisplayApplyNow = copyDisplayApplyNow.indexOf(true);
						copyDisplayApplyNow[indexOfDisplayApplyNow] = false;
						copyDisplayApplyNow[indexOfDisplayApplyNow - 1] = true;
						changeDisplayApplyNow(copyDisplayApplyNow);
						window.scrollTo({ top: 0, behavior: 'smooth' });
					}}
				>
					<img src={arrowLeft} alt='' className='applyNow__arrowLeft' />

					{i18n.t('back')}
				</button>

				<button
					type='button'
					className='applyNow__nextButton'
					onClick={() => {
						next();
						window.scrollTo({ top: 0, behavior: 'smooth' });
					}}
				>
					{i18n.t('next-step')}
					<img src={arrowRight} alt='' className='applyNow__arrowRight' />
				</button>
			</div>
		</section>
	);
};

const storeFuncs = {
	applyUser: funcsFromStore.applyUser,
	changeDisplayApplyNow: funcsFromStore.applyDisplayApplyNow,
};

const storeData = (state) => ({
	userFromRedux: state.user,
	displayApplyNow: state.displayApplyNow,
	languageSelected: state.language,
});

ApplyNow3.defaultProps = {
	languageSelected: '',
	displayApplyNow: [],
	changeDisplayApplyNow: () => {},
	userFromRedux: {},
	applyUser: () => {},
};

ApplyNow3.propTypes = {
	languageSelected: PropTypes.string,
	displayApplyNow: PropTypes.arrayOf(PropTypes.bool),
	changeDisplayApplyNow: PropTypes.func,
	userFromRedux: PropTypes.objectOf(PropTypes.any),
	applyUser: PropTypes.func,
};

export default connect(storeData, storeFuncs)(ApplyNow3);
