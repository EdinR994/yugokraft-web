    src/components/ApplyPage/TermsAndConditions.js

    import './styles.scss';
import React from 'react';
import i18n from 'i18n-js';
import arrowLeft from './images/arrowLeft.svg';

/**
 * @desc UI for terms and conditions
 * @param {object} displayApplyNow
 * @param {function} changeDisplayApplyNow
 * @returns {JSX.Element|null}
 */
const TermsAndConditions = ({ displayApplyNow, changeDisplayApplyNow }) => {
	if (!displayApplyNow[7]) return null;
	window.scrollTo({ top: 0, behavior: 'smooth' });

	return (
		<section className='applyNow__page'>
			<form className='applyNow__form applyNow__form5'>
				<h2 className='applyNow__header'>{i18n.t('GDPR-Terms-and-Conditions')}</h2>

				<div className='applyNow__TermsAndConditionsWrapper'>
					{i18n.t('terms-and-conditions')}
					<br />
					<br />
					<h2 className='applyNow__addText'>{i18n.t('welcome-to-yugocraft')}</h2>
					<br />
					<br />
					{i18n.t('this-terms-and-conditions')}
					<br />
					<br />
					{i18n.t('by-accessing-this-website')}
					<br />
					<br />
					{i18n.t('The-following-terminology-appliese')}
				</div>
			</form>

			<div className='applyNow__labelsWrapper applyNow__ButtonsWrapper2'>
				<button
					type='button'
					className='applyNow__backButton'
					onClick={() => {
						const copyArr = [...displayApplyNow];
						const indexOfDisplayApplyNow = copyArr.indexOf(true);
						copyArr[indexOfDisplayApplyNow] = false;
						copyArr[indexOfDisplayApplyNow - 1] = true;
						changeDisplayApplyNow(copyArr);
					}}
				>
					<img src={arrowLeft} alt='' className='applyNow__arrowLeft' />

					{i18n.t('back')}
				</button>

				<button
					type='button'
					className='applyNow__nextButton applyNow__sendAppButton'
					onClick={() => {
						const copyArr = [...displayApplyNow];
						const indexOfDisplayApplyNow = copyArr.indexOf(true);
						copyArr[indexOfDisplayApplyNow] = false;
						copyArr[indexOfDisplayApplyNow + 1] = true;
						changeDisplayApplyNow(copyArr);
					}}
				>
					{i18n.t('next-step')}
				</button>
			</div>
		</section>
	);
};

export default TermsAndConditions;
